-- AdminModule.lua
-- This module sets up RemoteEvents for admin functionalities and listens for admin requests/commands.
-- It now fully implements Kick, Bring, Teleport, Kill, GodMode, Speed,
-- and it toggles Fly by firing a "FlyClient" event to the target's client.
--
-- Note: For the fly command, the server simply signals the target to toggle their local fly mode.
local AdminModule = {}
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Players = game:GetService("Players")

function AdminModule.Initialize()
	-- Create or find the folder that holds our RemoteEvents.
	local remoteFolder = ReplicatedStorage:FindFirstChild("AdminRemoteEvents")
	if not remoteFolder then
		remoteFolder = Instance.new("Folder")
		remoteFolder.Name = "AdminRemoteEvents"
		remoteFolder.Parent = ReplicatedStorage
		print("AdminModule: Created AdminRemoteEvents folder.")
	end

	local requestAdmin = remoteFolder:FindFirstChild("RequestAdmin")
	if not requestAdmin then
		requestAdmin = Instance.new("RemoteEvent")
		requestAdmin.Name = "RequestAdmin"
		requestAdmin.Parent = remoteFolder
		print("AdminModule: Created RequestAdmin RemoteEvent.")
	end

	local adminCommand = remoteFolder:FindFirstChild("AdminCommand")
	if not adminCommand then
		adminCommand = Instance.new("RemoteEvent")
		adminCommand.Name = "AdminCommand"
		adminCommand.Parent = remoteFolder
		print("AdminModule: Created AdminCommand RemoteEvent.")
	end

	local consoleUpdate = remoteFolder:FindFirstChild("ConsoleUpdate")
	if not consoleUpdate then
		consoleUpdate = Instance.new("RemoteEvent")
		consoleUpdate.Name = "ConsoleUpdate"
		consoleUpdate.Parent = remoteFolder
		print("AdminModule: Created ConsoleUpdate RemoteEvent.")
	end

	local flyClient = remoteFolder:FindFirstChild("FlyClient")
	if not flyClient then
		flyClient = Instance.new("RemoteEvent")
		flyClient.Name = "FlyClient"
		flyClient.Parent = remoteFolder
		print("AdminModule: Created FlyClient RemoteEvent.")
	end

	local function sendConsoleUpdate(player, message)
		consoleUpdate:FireClient(player, message)
		print("AdminModule: Sent console update to " .. player.Name .. ": " .. message)
	end

	local function processAdminCommand(player, command, targetName, extra)
		local target = Players:FindFirstChild(targetName)
		if not target or not target.Character then
			sendConsoleUpdate(player, "Player '" .. targetName .. "' not found or has no character.")
			return
		end

		print("AdminModule: Processing command: " .. command .. " for " .. targetName .. " from " .. player.Name)
		if command == "Kick" then
			target:Kick("Kicked by admin command.")
			sendConsoleUpdate(player, "Player '" .. targetName .. "' has been kicked.")
		elseif command == "Bring" then
			if target.Character and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
				local pos = player.Character.HumanoidRootPart.Position + Vector3.new(5, 0, 0)
				target.Character:MoveTo(pos)
				sendConsoleUpdate(player, "Brought player '" .. targetName .. "' to you.")
			else
				sendConsoleUpdate(player, "Unable to bring '" .. targetName .. "'.")
			end
		elseif command == "Teleport" then
			if target.Character and player.Character and target.Character:FindFirstChild("HumanoidRootPart") then
				local pos = target.Character.HumanoidRootPart.CFrame + Vector3.new(5, 0, 0)
				player.Character:SetPrimaryPartCFrame(pos)
				sendConsoleUpdate(player, "Teleported to '" .. targetName .. "'.")
			else
				sendConsoleUpdate(player, "Unable to teleport to '" .. targetName .. "'.")
			end
		elseif command == "Kill" then
			if target.Character and target.Character:FindFirstChild("Humanoid") then
				target.Character.Humanoid.Health = 0
				sendConsoleUpdate(player, "Killed '" .. targetName .. "'.")
			else
				sendConsoleUpdate(player, "Unable to kill '" .. targetName .. "'.")
			end
		elseif command == "GodMode" then
			if target.Character then
				local humanoid = target.Character:FindFirstChildOfClass("Humanoid")
				if humanoid then
					local godModeTag = target.Character:FindFirstChild("GodModeEnabled")
					if godModeTag then
						-- Disable God Mode
						godModeTag:Destroy()
						humanoid.MaxHealth = 100
						if humanoid.Health > 100 then
							humanoid.Health = 100
						end
						sendConsoleUpdate(player, "Disabled God Mode for '" .. targetName .. "'.")
					else
						-- Enable God Mode
						local tag = Instance.new("BoolValue")
						tag.Name = "GodModeEnabled"
						tag.Parent = target.Character
						humanoid.MaxHealth = 99999
						humanoid.Health = 99999
						sendConsoleUpdate(player, "Enabled God Mode for '" .. targetName .. "'.")
					end
				else
					sendConsoleUpdate(player, "No humanoid found for '" .. targetName .. "'.")
				end
			end
		elseif command == "Fly" then
			-- Instead of manipulating physics on the server, signal the target's client.
			local flyTag = target.Character:FindFirstChild("FlyEnabled")
			if flyTag then
				flyTag:Destroy()
				flyClient:FireClient(target, "Disable")
				sendConsoleUpdate(player, "Disabled Fly for '" .. targetName .. "'.")
			else
				local tag = Instance.new("BoolValue")
				tag.Name = "FlyEnabled"
				tag.Parent = target.Character
				flyClient:FireClient(target, "Enable")
				sendConsoleUpdate(player, "Enabled Fly for '" .. targetName .. "'.")
			end
		elseif command == "Speed" then
			if target.Character then
				local humanoid = target.Character:FindFirstChildOfClass("Humanoid")
				if humanoid then
					local newSpeed = tonumber(extra) or 16
					humanoid.WalkSpeed = newSpeed
					sendConsoleUpdate(player, "Set speed for '" .. targetName .. "' to " .. humanoid.WalkSpeed .. ".")
				else
					sendConsoleUpdate(player, "No humanoid found for speed command on '" .. targetName .. "'.")
				end
			end
		else
			sendConsoleUpdate(player, "Unknown command: " .. command)
		end
	end

	requestAdmin.OnServerEvent:Connect(function(player)
		print("AdminModule: Received RequestAdmin from " .. player.Name)
		sendConsoleUpdate(player, "Admin request received. Activating admin panel.")
		adminCommand:FireClient(player, "Activate")
		print("AdminModule: Fired Activate command to " .. player.Name)
	end)

	adminCommand.OnServerEvent:Connect(function(player, command, targetName, extra)
		processAdminCommand(player, command, targetName, extra)
	end)

	print("AdminModule: Admin module initialized.")
end

return AdminModule
