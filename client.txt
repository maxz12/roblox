-- AdminClientScript.lua
-- Place this LocalScript in StarterPlayerScripts.
-- This client script creates a modern, movable admin GUI with two dynamic columns:
--  • Left: A list of players (auto-refresh on join/leave) where you can select any player.
--         The selected player's row is highlighted with a distinct color.
--  • Right: A commands panel with dynamic sliders and toggle buttons which reflect the
--         enabled/disabled states and current settings of the selected player.
--
-- Commands can be executed for all players. For the LocalPlayer, some commands run locally
-- (e.g. Fly and Invisibility) while others (e.g. Speed and JumpPower adjustments) update both
-- the client and send the remote event. For remote players, all commands are fired via the remote event.
--
-- Commands implemented:
--  • Fly: Toggles traditional flying (using the mouse's hit position for direction).
--  • Invisibility: Toggles the character's visibility.
--  • JumpPower & Speed: Adjust via sliders.
--  • Fling: Applies a temporary angular impulse.
--
-- For remote players, a remote event (adminCommand:FireServer) is fired with an extra argument
-- indicating the new toggle state where applicable.
--

local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local player = Players.LocalPlayer
local mouse = player:GetMouse()

-- Ensure PlayerGui is ready.
local PlayerGui = player:WaitForChild("PlayerGui")

-- RemoteEvents from the AdminRemoteEvents folder.
local remoteFolder = ReplicatedStorage:WaitForChild("AdminRemoteEvents")
local requestAdmin = remoteFolder:WaitForChild("RequestAdmin")
local adminCommand = remoteFolder:WaitForChild("AdminCommand")
local consoleUpdate = remoteFolder:WaitForChild("ConsoleUpdate")
local flyClient = remoteFolder:WaitForChild("FlyClient")

-- UI Theme settings.
local UITheme = {
	BackgroundColor = Color3.fromRGB(30, 30, 30),
	AccentColor = Color3.fromRGB(45, 45, 45),
	TextColor = Color3.fromRGB(235, 235, 235),
	ButtonColor = Color3.fromRGB(50, 150, 250),
	ToggledButtonColor = Color3.fromRGB(30, 100, 200),
	HighlightColor = Color3.fromRGB(255, 165, 0) -- distinct highlight for selected player.
}

-- Global state for local commands.
local localFlyActive = false
local localInvisibleActive = false

-- Default settings.
local defaultSpeed = 16
local defaultJumpPower = 50

-- Global variable to track the currently selected player.
local selectedPlayer = player

-- Table to store command states per player.
-- Each player state includes:
--   fly      -- whether fly is enabled.
--   invisible-- whether invisibility is enabled.
--   speed    -- current WalkSpeed.
--   jump     -- current JumpPower.
local playerStates = {}

local function initializePlayerState(p)
	local state = {
		fly = false,
		invisible = false,
		speed = defaultSpeed,
		jump = defaultJumpPower
	}
	if p == player then
		local character = p.Character
		if character and character:FindFirstChildOfClass("Humanoid") then
			state.speed = character:FindFirstChildOfClass("Humanoid").WalkSpeed or defaultSpeed
			state.jump = character:FindFirstChildOfClass("Humanoid").JumpPower or defaultJumpPower
		end
	end
	return state
end

-- Initialize states for existing players.
for _, p in ipairs(Players:GetPlayers()) do
	playerStates[p] = initializePlayerState(p)
end

-- Continuously update local player's speed and jump based on current character settings.
RunService.RenderStepped:Connect(function()
	if player.Character and player.Character:FindFirstChildOfClass("Humanoid") then
		playerStates[player].speed = player.Character:FindFirstChildOfClass("Humanoid").WalkSpeed
		playerStates[player].jump = player.Character:FindFirstChildOfClass("Humanoid").JumpPower
	end
end)

-- Local Command Implementations for the LocalPlayer

-- Invisibility: Toggle transparency for all BaseParts (except HumanoidRootPart).
local function toggleInvisibility(char)
	for _, part in ipairs(char:GetDescendants()) do
		if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
			part.Transparency = localInvisibleActive and 0 or 1
		end
	end
	-- Toggle state.
	localInvisibleActive = not localInvisibleActive
	playerStates[player].invisible = localInvisibleActive
	AppendConsole("Invisibility " .. (localInvisibleActive and "enabled" or "disabled") .. " for " .. player.Name)
end

-- Traditional Fly (local): use mouse hit for direction.
local localFlyConnection, localFlyVelocity, localFlyGyro

local function startLocalFly()
	local character = player.Character or player.CharacterAdded:Wait()
	local hrp = character:WaitForChild("HumanoidRootPart")
	-- Create BodyVelocity and BodyGyro.
	localFlyVelocity = Instance.new("BodyVelocity", hrp)
	localFlyVelocity.MaxForce = Vector3.new(1e5, 1e5, 1e5)
	localFlyVelocity.Velocity = Vector3.new(0, 0, 0)

	localFlyGyro = Instance.new("BodyGyro", hrp)
	localFlyGyro.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
	localFlyGyro.P = 1000

	localFlyActive = true
	playerStates[player].fly = true
	AppendConsole("Fly enabled for " .. player.Name)

	-- Update movement based on mouse.Hit.
	localFlyConnection = RunService.RenderStepped:Connect(function()
		local hrpPos = hrp.Position
		local targetPos = mouse.Hit.p
		local direction = (targetPos - hrpPos)
		if direction.Magnitude > 0 then
			localFlyVelocity.Velocity = direction.Unit * 50
			localFlyGyro.CFrame = CFrame.new(hrpPos, hrpPos + direction)
		else
			localFlyVelocity.Velocity = Vector3.new(0, 0, 0)
		end
	end)
end

local function stopLocalFly()
	if localFlyConnection then
		localFlyConnection:Disconnect()
		localFlyConnection = nil
	end
	if localFlyVelocity then localFlyVelocity:Destroy() end
	if localFlyGyro then localFlyGyro:Destroy() end
	localFlyActive = false
	playerStates[player].fly = false
	AppendConsole("Fly disabled for " .. player.Name)
end

-- Fling: Applies a temporary BodyAngularVelocity for a fling effect.
local function flingCharacter(char)
	local hrp = char:FindFirstChild("HumanoidRootPart")
	if hrp then
		local bav = Instance.new("BodyAngularVelocity", hrp)
		bav.AngularVelocity = Vector3.new(20, 20, 20)
		bav.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
		delay(0.5, function()
			if bav then bav:Destroy() end
		end)
		AppendConsole("Fling executed for " .. player.Name)
	end
end

-- Function to refresh the players list.
local function RefreshPlayersList()
	-- Clear existing player rows.
	for _, child in ipairs(playersScroll:GetChildren()) do
		if child:IsA("TextButton") then
			child:Destroy()
		end
	end

	-- Ensure state exists for every player.
	for _, p in ipairs(Players:GetPlayers()) do
		if not playerStates[p] then
			playerStates[p] = initializePlayerState(p)
		end
	end

	-- Get all players, with the local player at the top.
	local allPlayers = Players:GetPlayers()
	table.sort(allPlayers, function(a, b)
		if a == player then return true end
		if b == player then return false end
		return a.Name < b.Name
	end)

	-- Create a button for each player.
	for _, p in ipairs(allPlayers) do
		local row = Instance.new("TextButton")
		row.Name = "PlayerRow_" .. p.Name
		row.Size = UDim2.new(1, 0, 0, 30)
		-- Always highlight the selected player with a distinct color.
		row.BackgroundColor3 = (p == selectedPlayer) and UITheme.HighlightColor or UITheme.AccentColor
		row.TextColor3 = UITheme.TextColor
		row.Font = Enum.Font.Gotham
		row.TextScaled = true
		row.Text = p.Name
		row.Parent = playersScroll

		row.MouseButton1Click:Connect(function()
			selectedPlayer = p
			-- Update selected player label.
			selectedPlayerLabel.Text = "Selected: " .. selectedPlayer.Name
			RefreshPlayersList()
		end)
	end
end

-- Dynamic Sliders: Create sliders that return their UI elements.
local function createSlider(parent, labelText, initialValue, minValue, maxValue, callback, posY)
	local container = Instance.new("Frame")
	container.Size = UDim2.new(1, -10, 0, 40)
	container.Position = UDim2.new(0, 5, 0, posY)
	container.BackgroundTransparency = 1
	container.Parent = parent

	local sliderLabel = Instance.new("TextLabel")
	sliderLabel.Name = labelText.."Label"
	sliderLabel.Size = UDim2.new(1, 0, 0, 20)
	sliderLabel.BackgroundTransparency = 1
	sliderLabel.TextColor3 = UITheme.TextColor
	sliderLabel.Font = Enum.Font.Gotham
	sliderLabel.TextScaled = true
	sliderLabel.Text = labelText .. ": " .. tostring(initialValue)
	sliderLabel.Parent = container

	local sliderBg = Instance.new("Frame")
	sliderBg.Name = labelText.."SliderBg"
	sliderBg.Size = UDim2.new(1, 0, 0, 10)
	sliderBg.Position = UDim2.new(0, 0, 0, 25)
	sliderBg.BackgroundColor3 = UITheme.AccentColor
	sliderBg.BorderSizePixel = 0
	sliderBg.Parent = container

	local sliderFill = Instance.new("Frame")
	sliderFill.Name = labelText.."SliderFill"
	sliderFill.Size = UDim2.new((initialValue - minValue) / (maxValue - minValue), 0, 1, 0)
	sliderFill.BackgroundColor3 = UITheme.ButtonColor
	sliderFill.BorderSizePixel = 0
	sliderFill.Parent = sliderBg

	local sliderHandle = Instance.new("TextButton")
	sliderHandle.Name = labelText.."SliderHandle"
	sliderHandle.Size = UDim2.new(0, 18, 1, 0)
	sliderHandle.BackgroundColor3 = UITheme.TextColor
	sliderHandle.Text = ""
	sliderHandle.BorderSizePixel = 0
	sliderHandle.Parent = sliderBg
	sliderHandle.Position = UDim2.new((initialValue - minValue) / (maxValue - minValue), -9, 0, 0)

	local dragging = false
	local function updateSlider(posX)
		local relative = math.clamp(posX / sliderBg.AbsoluteSize.X, 0, 1)
		local value = minValue + relative * (maxValue - minValue)
		sliderFill.Size = UDim2.new(relative, 0, 1, 0)
		sliderHandle.Position = UDim2.new(relative, -9, 0, 0)
		sliderLabel.Text = labelText .. ": " .. tostring(math.floor(value))
		callback(math.floor(value))
	end

	sliderHandle.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = true
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
			local pos = sliderBg.AbsolutePosition.X
			local x = input.Position.X - pos
			updateSlider(x)
		end
	end)

	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = false
		end
	end)

	return {
		container = container,
		label = sliderLabel,
		bg = sliderBg,
		fill = sliderFill,
		handle = sliderHandle,
		min = minValue,
		max = maxValue
	}
end

-- Create Speed slider UI.

	-- For local player, update character; for remote, fire event.

			player.Character:FindFirstChildOfClass("Humanoid").WalkSpeed = value
		end

-- List of command buttons.
local commandsList = {
	"Kick",
	"Bring",
	"Teleport",
	"Kill",
	"GodMode",
	"Fly",
	"Heal",
	"Fling",
	"Invisible"
}

local commandButtonRefs = {}

local function SetupCommandButtons()
	-- Clear existing buttons.
	for _, child in ipairs(cmdsGridParent:GetChildren()) do
		if child:IsA("TextButton") then
			child:Destroy()
		end
	end
	commandButtonRefs = {}
	-- Create a button for each command.
	for _, cmd in ipairs(commandsList) do
		local button = Instance.new("TextButton")
		button.Name = cmd.."Button"
		button.Text = cmd
		button.BackgroundColor3 = UITheme.ButtonColor
		button.TextColor3 = UITheme.TextColor
		button.Font = Enum.Font.GothamSemibold
		button.TextScaled = true
		button.Parent = cmdsGridParent
		commandButtonRefs[cmd] = button

		button.Activated:Connect(function()
			-- When executing a command, check if selected player is local.
			if selectedPlayer == player then
				-- Local execution.
				if cmd == "Fly" then
					if localFlyActive then
						stopLocalFly()
					else
						startLocalFly()
					end
					button.BackgroundColor3 = localFlyActive and UITheme.ToggledButtonColor or UITheme.ButtonColor
				elseif cmd == "Invisible" then
					if player.Character then
						toggleInvisibility(player.Character)
					end
					button.BackgroundColor3 = localInvisibleActive and UITheme.ToggledButtonColor or UITheme.ButtonColor
				elseif cmd == "Fling" then
					if player.Character then
						flingCharacter(player.Character)
					end
				else
					adminCommand:FireServer(cmd, selectedPlayer.Name)
					AppendConsole(cmd .. " command fired on " .. selectedPlayer.Name)
				end
			else
				-- For remote players, toggle the state for togglable commands.
				if cmd == "Fly" then
					local newState = not playerStates[selectedPlayer].fly
					playerStates[selectedPlayer].fly = newState
					adminCommand:FireServer("Fly", selectedPlayer.Name, newState)
					AppendConsole("Fly " .. (newState and "enabled" or "disabled") .. " on " .. selectedPlayer.Name)
				elseif cmd == "Invisible" then
					local newState = not playerStates[selectedPlayer].invisible
					playerStates[selectedPlayer].invisible = newState
					adminCommand:FireServer("Invisible", selectedPlayer.Name, newState)
					AppendConsole("Invisible " .. (newState and "enabled" or "disabled") .. " on " .. selectedPlayer.Name)
				elseif cmd == "Fling" then
					adminCommand:FireServer("Fling", selectedPlayer.Name)
					AppendConsole("Fling executed on " .. selectedPlayer.Name)
				else
					adminCommand:FireServer(cmd, selectedPlayer.Name)
					AppendConsole(cmd .. " command fired on " .. selectedPlayer.Name)
				end
			end
		end)
	end
end

SetupCommandButtons()

-- Dynamic UI Update: Refresh sliders and toggle button appearance based on player's state.
local function updateDynamicUI()
	-- Update selected player label.
	selectedPlayerLabel.Text = "Selected: " .. selectedPlayer.Name

	-- Get state for the selected player.
	local state = playerStates[selectedPlayer]
	if not state then
		state = initializePlayerState(selectedPlayer)
		playerStates[selectedPlayer] = state
	end

	-- Update sliders.
	local speedVal = state.speed
	local sp = speedSliderUI
	local relSpeed = math.clamp((speedVal - sp.min) / (sp.max - sp.min), 0, 1)
	sp.fill.Size = UDim2.new(relSpeed, 0, 1, 0)
	sp.handle.Position = UDim2.new(relSpeed, -9, 0, 0)
	sp.label.Text = "Speed: " .. tostring(speedVal)

	local jumpVal = state.jump
	local jp = jumpSliderUI
	local relJump = math.clamp((jumpVal - jp.min) / (jp.max - jp.min), 0, 1)
	jp.fill.Size = UDim2.new(relJump, 0, 1, 0)
	jp.handle.Position = UDim2.new(relJump, -9, 0, 0)
	jp.label.Text = "JumpPower: " .. tostring(jumpVal)

	-- Update toggle button appearance.
	if commandButtonRefs["Fly"] then
		commandButtonRefs["Fly"].BackgroundColor3 = state.fly and UITheme.ToggledButtonColor or UITheme.ButtonColor
	end
	if commandButtonRefs["Invisible"] then
		commandButtonRefs["Invisible"].BackgroundColor3 = state.invisible and UITheme.ToggledButtonColor or UITheme.ButtonColor
	end
end

RunService.RenderStepped:Connect(function()
	updateDynamicUI()
end)
