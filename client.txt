-- Command button creation helper
local function createCommandButton(name)
    local button = Instance.new("TextButton")
    button.BackgroundColor3 = UITheme.ButtonColor
    button.TextColor3 = UITheme.TextColor
    button.Text = name
    button.Font = Enum.Font.GothamSemibold
    button.TextSize = 14
    button.Parent = commandsPanel
    return button
end

-- Create command buttons
local commands = {
    "Kick", "Bring", "Teleport", "Kill", "GodMode",
    "Fly", "Heal", "Fling", "Invisible"
}

local buttons = {}
for _, cmd in ipairs(commands) do
    buttons[cmd] = createCommandButton(cmd)

    buttons[cmd].MouseButton1Click:Connect(function()
        if cmd == "Fly" then
            local state = not playerStates[selectedPlayer].fly
            playerStates[selectedPlayer].fly = state
            if selectedPlayer == player then
                if state then
                    enableFly()
                else
                    disableFly()
                end
            else
                adminCommand:FireServer(cmd, selectedPlayer.Name, state)
            end
            buttons[cmd].BackgroundColor3 = state and UITheme.ToggledButtonColor or UITheme.ButtonColor
        elseif cmd == "Invisible" then
            local state = not playerStates[selectedPlayer].invisible
            playerStates[selectedPlayer].invisible = state
            adminCommand:FireServer(cmd, selectedPlayer.Name, state)
            buttons[cmd].BackgroundColor3 = state and UITheme.ToggledButtonColor or UITheme.ButtonColor
        else
            adminCommand:FireServer(cmd, selectedPlayer.Name)
        end
    end)
end

-- Handle fly command locally
local function enableFly()
    local char = player.Character
    if not char then return end

    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    if flyBodyVelocity then flyBodyVelocity:Destroy() end
    if flyBodyGyro then flyBodyGyro:Destroy() end

    flyBodyVelocity = Instance.new("BodyVelocity")
    flyBodyVelocity.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    flyBodyVelocity.Velocity = Vector3.new(0, 0, 0)
    flyBodyVelocity.Parent = hrp

    flyBodyGyro = Instance.new("BodyGyro")
    flyBodyGyro.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
    flyBodyGyro.P = 1000
    flyBodyGyro.Parent = hrp

    localFlyActive = true

    if flyConnection then flyConnection:Disconnect() end
    flyConnection = RunService.RenderStepped:Connect(function()
        if not localFlyActive then return end

        local moveDirection = Vector3.new(0, 0, 0)

        if UserInputService:IsKeyDown(Enum.KeyCode.W) then
            moveDirection = moveDirection + workspace.CurrentCamera.CFrame.LookVector
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then
            moveDirection = moveDirection - workspace.CurrentCamera.CFrame.LookVector
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then
            moveDirection = moveDirection - workspace.CurrentCamera.CFrame.RightVector
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then
            moveDirection = moveDirection + workspace.CurrentCamera.CFrame.RightVector
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
            moveDirection = moveDirection + Vector3.new(0, 1, 0)
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
            moveDirection = moveDirection - Vector3.new(0, 1, 0)
        end

        if moveDirection.Magnitude > 0 then
            flyBodyVelocity.Velocity = moveDirection.Unit * 50
        else
            flyBodyVelocity.Velocity = Vector3.new(0, 0, 0)
        end

        flyBodyGyro.CFrame = workspace.CurrentCamera.CFrame
    end)
end

local function disableFly()
    localFlyActive = false
    if flyConnection then
        flyConnection:Disconnect()
        flyConnection = nil
    end
    if flyBodyVelocity then
        flyBodyVelocity:Destroy()
        flyBodyVelocity = nil
    end
    if flyBodyGyro then
        flyBodyGyro:Destroy()
        flyBodyGyro = nil
    end
end

-- Update players list
local function updatePlayersList()
    for _, child in ipairs(playersList:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end

    for _, p in ipairs(Players:GetPlayers()) do
        if not playerStates[p] then
            playerStates[p] = initPlayerState(p)
        end

        local button = Instance.new("TextButton")
        button.Size = UDim2.new(1, -10, 0, 25)
        button.BackgroundColor3 = p == selectedPlayer and UITheme.HighlightColor or UITheme.ButtonColor
        button.TextColor3 = UITheme.TextColor
        button.Text = p.Name
        button.TextSize = 14
        button.Font = Enum.Font.Gotham
        button.Parent = playersList

        button.MouseButton1Click:Connect(function()
            selectedPlayer = p
            updatePlayersList()
        end)
    end
end

-- Speed control
speedValue.FocusLost:Connect(function(enterPressed)
    if enterPressed then
        local speed = tonumber(speedValue.Text)
        if speed then
            adminCommand:FireServer("Speed", selectedPlayer.Name, speed)
        end
    end
end)

-- Handle remote events
flyClient.OnClientEvent:Connect(function(action)
    if action == "Enable" then
        enableFly()
    else
        disableFly()
    end
end)

consoleUpdate.OnClientEvent:Connect(function(message)
    consoleText.Text = consoleText.Text .. "\n" .. message
end)

-- Player added/removed handlers
Players.PlayerAdded:Connect(function(p)
    playerStates[p] = initPlayerState(p)
    updatePlayersList()
end)

Players.PlayerRemoving:Connect(function(p)
    playerStates[p] = nil
    if selectedPlayer == p then
        selectedPlayer = player
    end
    updatePlayersList()
end)

-- Initialize
updatePlayersList()
requestAdmin:FireServer()

-- Clean up
player.CharacterRemoving:Connect(disableFly)
